for i in range(len(r)):
    # Input file for ORCA
    with open("h2_scan.in", "w") as input_file:
        text = f'''! B3LYP cc-pVQZ TightSCF
        * xyz 0 1
        H 0.000000 0.000000 0.000000
        H {r[i]} 0.000000 0.000000
        *
        '''
        input_file.write(text)
    
    # Orca call

    # Output energy extraction

                # Subtract the atomic energy of each hydrogen atom 2*13.6 eV
                e += 2*13.6
                # Save energy in the array
                energy[i] = e
                # Print partial results
                print(f"{i}. r = {r[i]}, E = {energy[i]} eV")

# Fit Morse potential
def morse(r, D, a, r0, c):
    return D * (1 - np.exp(-a * (r - r0)))**2 - c
# Initial guess for the parameters

# Fit the Morse potential to the data
def fit_morse(r, energy):
    popt, _ = curve_fit(morse, r, energy, p0=[5, 2, r_eq, E_min])
    return popt
D, a, r0, c = fit_morse(r, energy)
print(f"Fitted parameters: D = {D}, a = {a}, r0 = {r0}, c = {c}")

# First derivative of Morse potential
def morse_first_derivative(r, D, a, r0):
    return 2 * D * a * (1 - np.exp(-a * (r - r0))) * np.exp(-a * (r - r0))

# Find the minimum
r_eq_fit = fsolve(morse_first_derivative, r, args=(D, a, r0), xtol=1e-4)[0]
E_min_fit = morse(r_eq_fit, D, a, r0, c)
print(f"Fit Equilibrium: {r_eq_fit} Angstrom")
print(f"Fit Minimum Energy: {E_min_fit} eV")

La constante de fuerza $k(r)$ se obtiene como la segunda derivada de la curva de energa. Se calcula utilizando la segunda derivada analtica del potencial de Morse: $$k_{fit}(r)=\frac{d^2E_{fit}(r)}{dr^2}$$

Se evala la segunda derivada en el punto de equilibrio $k_{fit}(r_{eq,fit})$.

Adems, se incluye el clculo de la segunda derivada numrica en el punto de equilibrio, lo que permite estimar el error de interpolacin: $$k(r_{eq})=\frac{E(r_{eq+1}) - 2E(r_{eq}) + E(r_{eq-1})}{r_{eq}-r_{eq-1}}$$

# Numerical second derivative through finite differences
def numerical_second_derivative(r, energy, index_min):
    h = r[1] - r[0]
    d2E = (energy[index_min + 1] - 2 * energy[index_min] + energy[index_min - 1]) / (h**2)
    return d2E

# Morse potential second derivative
def morse_second_derivative(r, D, a, r0):
    return 2 * D * a**2 * np.exp(-2 * a * (r - r0))

# Find the second derivative at the minimum
k_eq = numerical_second_derivative(r, energy, index_min)
k_eq_fit = morse_second_derivative(r_eq_fit, D, a, r0)
